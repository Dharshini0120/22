import { createSlice, PayloadAction } from '@reduxjs/toolkit';

interface PasswordState {
  lastPasswordChanged: string | null;
  loading: boolean;
  error: string | null;
}

const initialState: PasswordState = {
  lastPasswordChanged: null,
  loading: false,
  error: null,
};

const passwordSlice = createSlice({
  name: 'password',
  initialState,
  reducers: {
    setLastPasswordChanged: (state, action: PayloadAction<string>) => {
      state.lastPasswordChanged = action.payload;
      state.error = null;
    },
    setPasswordLoading: (state, action: PayloadAction<boolean>) => {
      state.loading = action.payload;
    },
    setPasswordError: (state, action: PayloadAction<string | null>) => {
      state.error = action.payload;
      state.loading = false;
    },
    clearPasswordState: (state) => {
      state.lastPasswordChanged = null;
      state.loading = false;
      state.error = null;
    },
  },
});

export const { 
  setLastPasswordChanged, 
  setPasswordLoading, 
  setPasswordError, 
  clearPasswordState 
} = passwordSlice.actions;

export default passwordSlice.reducer;
