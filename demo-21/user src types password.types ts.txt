// User Portal Password Change Input
export interface ChangePasswordInput {
  userId: string;
  oldPassword: string;
  newPassword: string;
  confirmPassword: string;
}

// Password Change Response
export interface PasswordChangeResponse {
  status: string;
  message: string;
  statusCode: number;
  data: {
    userId: string;
    updatedAt: string;
  } | null;
  error: string | null;
}
